205/* Sparse matrix with up to 7 non-zero values per row */
210class SparseMat : public Mat {
211public:
217 SparseMat(size_t nu) : Mat{ nu } {
218 // memory for coefficients
219 a = new double*[nu];
220 for (size_t i = 0; i < nu; i++)
221 a[i] = new double[max_vals];
222
223 // memory for column indexes
224 c = new int*[nu];
225 for (size_t i = 0; i < nu; i++)
226 c[i] = new int[max_vals];
227
228 // clear data, column set to -1 are not set
229 for (size_t i = 0; i < nu; i++)
230 for (size_t j = 0; j < max_vals; j++) {
231 c[i][j] = -1;
232 a[i][j] = 0;
233 }
234 }
235
239 ~SparseMat() {
240 for (size_t i = 0; i < nu; i++)
241 delete[] a[i];
242 delete[] a;
243
244 for (size_t i = 0; i < nu; i++)
245 delete[] c[i];
246 delete[] c;
247 }

256 double& operator()(int i, int j) {
257 // search for the sparse column corresponding to full matrix column j
258 for (int v = 0; v < max_vals; v++) {
259 // did we reach an empty slot? If so, make it correspond to column j
260 if (c[i][v] < 0) c[i][v] = j;
261
262 // does this sparse column map to j? If so, return the coeff
263 if (c[i][v] == j) return a[i][v];
264 }
265
266 // getting here implies that all max_val slots are already occupied by columns other than j
267 throw std::runtime_error("Sparse matrix too small!");
268 return a[i][0]; // just to avoid a warning
269 }
